```python
# Homework Problem:
# Write a program to read through the mbox-short.txt and figure out the distribution by hour of the day for each
# of the messages. You can pull the hour out from the 'From ' line by finding the time and then splitting the string
# a second time using a colon.

# Return the solution with comments on new or altered lines. In comments, provide a summary of the changes:

times_emails_sent = list()

while True:
    try:
        file_name = input('Please enter a text file: ')
        if len(file_name) < 1:
            file_name = 'mbox-short.txt'
        with open(file_name) as file_handle:
            for line in file_handle:
                if line.startswith('From') and (len(line.split()) > 2):
                    # Extract the time from the 'From ' line and split it on the colon
                    time_email_sent = line.rstrip().split()[-2:-1][0]
                    # Split the time on the colon to get the hour
                    hour = time_email_sent.split(':')[0]
                    times_emails_sent.append(hour)

        break

    except FileNotFoundError:
        print('The file you entered does not exist. Please enter a valid file name.')

# Create a dictionary to store the distribution of emails by hour
hours_distribution = dict()

# Count the number of emails received for each hour
for hour in times_emails_sent:
    hours_distribution[hour] = hours_distribution.get(hour, 0) + 1

# Create a list of tuples representing the hour and count
hour_list = list()
for key, value in hours_distribution.items():
    hour_and_count = (key, value)
    hour_list.append(hour_and_count)

# Sort the list by hour
for hour_and_count in sorted(hour_list):
    hour = hour_and_count[0]
    count = hour_and_count[1]
    print(hour, count)
```
